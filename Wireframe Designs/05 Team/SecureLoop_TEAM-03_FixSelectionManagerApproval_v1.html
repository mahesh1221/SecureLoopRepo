# Restrict eks-node-role S3 access to ONLY payments-logs-prod bucket
resource "aws_iam_role_policy" "eks_node_s3" {
  name = "eks-node-s3-scoped"
  role = aws_iam_role.eks_node.id

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Effect   = "Allow"
      Action   = ["s3:GetObject", "s3:PutObject"]
      Resource = "arn:aws:s3:::payments-logs-prod/*"
    }]
  })
}
