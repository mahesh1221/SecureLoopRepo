AI Sidekick
For mysqli, use mysqli_prepare() with parameter binding. This separates SQL structure from data entirely, so even crafted input can't alter query logic.
// Vulnerable — current state $username = $_POST['username']; $query = "SELECT * FROM users WHERE username='$username'"; // Fixed — parameterised binding $stmt = mysqli_prepare($conn, "SELECT * FROM users WHERE username=?"); mysqli_stmt_bind_param($stmt, "s", $username); mysqli_stmt_execute($stmt); $result = mysqli_stmt_get_result($stmt);
